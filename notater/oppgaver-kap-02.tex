\begin{oppgave}
Vis at følgende påstander er sanne for alle matriser $M$, $N$ og~$L$:
\begin{punkt}
$M \roweq M$.
\end{punkt}
\begin{punkt}
Hvis $M \roweq N$, så: $N \roweq M$.
\end{punkt}
\begin{punkt}
Hvis $M \roweq L$ og $L \roweq N$, så: $M \roweq N$.
\end{punkt}
\end{oppgave}
